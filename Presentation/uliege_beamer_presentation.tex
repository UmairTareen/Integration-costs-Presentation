\begin{frame}[fragile]
  \frametitle{Conclusion}

  \textbf{Conclusion:}
    \begin{itemize}
      \item {Integration costs computations can be made in a simple way using existing modeling tools.}
      \item {Solar integrations costs are 55 Euros/MWh compared to 13 Euros/MWh for nuclear by 2050.}
      \item {Complementarity and the diversity of different renewable energy sources; when combined, integration costs remain marginal even above 90\% penetration.}
    \end{itemize}
